In[3]:
def lemmatization(texts, allowed_postags=["NOUN", "ADJ", "VERB", "ADV"]):
    nlp = spacy.load("en_core_web_sm", disable=["parser", "ner"])
    return [" ".join([token.lemma_ for token in nlp(text) if token.pos_ in allowed_postags]) for text in texts]

lemmatized_texts = lemmatization(data)

def gen_words(texts):
    return [gensim.utils.simple_preprocess(text, deacc=True) for text in texts]

data_words = gen_words(lemmatized_texts)

Bigrams Creation

A bigram is a sequence of two adjacent elements from a string of tokens, which are typically letters, syllables, or words. Bigrams can contain significant information when doing Natural Language Processing.

In[5]:
#BIGRAMS
bigram_phrases = gensim.models.Phrases(data_words, min_count=5, threshold=50)

bigram = gensim.models.phrases.Phraser(bigram_phrases)

def make_bigrams(texts):
    return [bigram[doc] for doc in texts]

data_bigrams = make_bigrams(data_words)

DF Filter

The DF filter drops from the corpus words that appears in so many texts that they may be insignificant in an information theory aspect.

#DF FILTER
id2word = corpora.Dictionary(data_bigrams)
texts = data_bigrams
corpus = [id2word.doc2bow(text) for text in texts]

tfidf = TfidfModel(corpus, id2word=id2word)

low_value=0.4
words = []
words_missing_in_tfidf = []

for i in range(0, len(corpus)):
    bow = corpus[i]
    tfidf_ids = [id for id,value in tfidf[bow]]
    bow_ids = [id for id, value in bow]
    hfreq_words = [id for id in bow_ids if tfidf.dfs[id]/len(corpus) > low_value]
    drops = hfreq_words + words_missing_in_tfidf
    for item in drops:
        words.append(id2word[item])
    words_missing_in_tfidf = [id for id in bow_ids if id not in tfidf_ids] # The words with tf-idf socre 0 will be missing

    new_bow = [b for b in bow if b[0] not in hfreq_words and b[0] not in words_missing_in_tfidf]       
    corpus[i] = new_bow
